Lagtp = function(x, k){
  n = length(x)
  x1 = matrix(0, n)
  for (i in (k+1):n) {
    x1[i] = x[i-k]
  }
  return(x1)

Lagt = function(x, k){
  n = length(x)
  x1 = matrix(0, n, k)
  for (j in 1:k){
    for (i in (j+1):n) {
      x1[i,j] = x[i-j]
    }
  }
  return(x1)
